    <script>
        // 1. Список преподавателей (без паролей, так как профиль пока не нужен, или можно добавить)
        const teachers = [
            { name: "Слобоздина Наталья Ивановна", img: "Слобоздина.jpg" },
            { name: "Никитина Дарья Владимировна", img: "Никитина.jpg" },
            { name: "Рейтер Елена Анатольевна", img: "Рейтер.jpg" },
            { name: "Бормотова Елена Сергеевна", img: "Бормотова.jpg" },
            { name: "Куличёва Галина Васильевна", img: "Галина.jpg" },
            { name: "Новиков Денис Васильевич", img: "Новиков.jpg" },
            { name: "Сахарова Алла Михайловна", img: "Ала.jpg" }, 
            { name: "Рамазанов Рамазан Маратович", img: "Рамазанов.jpg" },
            { name: "Буленко Мария Валерьевна", img: "Буленко.jpg" },
            { name: "Алпатова Ирина Дмитриевна", img: "Алпатова.jpg" },
            { name: "Сазонова Вероника Сергеевна", img: "Сазонова.jpg" },
            { name: "Качурин Игорь Александрович", img: "Качурин.jpg" }
        ];

        // Пароли для преподавателей (если вдруг захочешь сделать вход и туда)
        // Пока они не используются в логике "Оставить отзыв", но могут пригодиться
        const teacherPasswords = {
            "Слобоздина Наталья Ивановна": "slobodzina_math",
            "Никитина Дарья Владимировна": "nikitina_eng",
            // ... можно добавить остальные, если нужно
        };

        const list = document.getElementById('teachers-list');
        const modal = document.getElementById('passwordModal');
        const modalName = document.getElementById('modal-teacher-name');
        const passInput = document.getElementById('pass-input');
        const submitBtn = document.getElementById('modal-btn-submit');
        const cancelBtn = document.getElementById('modal-btn-cancel');
        const errorMsg = document.getElementById('modal-error');

        let currentTeacherName = ""; 

        // Функция открытия модального окна
        function openPasswordModal(name) {
            currentTeacherName = name;
            modalName.textContent = `Вход: ${name}`;
            passInput.value = '';
            errorMsg.style.display = 'none';
            passInput.style.borderColor = '#ddd';
            modal.style.display = 'flex';
        }

        // Закрытие модального окна
        cancelBtn.onclick = () => { modal.style.display = 'none'; };

        // Проверка пароля (если преподаватель захочет зайти в свой личный кабинет)
        submitBtn.onclick = () => {
            const enteredPass = passInput.value;
            const correctPass = teacherPasswords[currentTeacherName];

            if (correctPass && enteredPass === correctPass) {
                // Если пароль верный, ведем на страницу профиля преподавателя
                const encodedName = encodeURIComponent(currentTeacherName);
                window.location.href = `teacher-profile.html?teacher=${encodedName}`;
            } else if (!correctPass) {
                alert("У этого преподавателя пока нет личного кабинета (или пароля).");
                modal.style.display = 'none';
            } else {
                errorMsg.style.display = 'block';
                passInput.style.borderColor = 'red';
                setTimeout(() => {
                    passInput.value = ''; passInput.style.borderColor = '#ddd'; errorMsg.style.display = 'none';
                }, 2000);
            }
        };

        // Генерация карточек преподавателей
        teachers.forEach(teacher => {
            const card = document.createElement('div');
            card.className = 'card';

            const hasImage = teacher.img && teacher.img !== "--------------" && teacher.img !== "";
            let imageHtml = hasImage ? `<img src="${teacher.img}" alt="${teacher.name}" onerror="this.style.display='none'">` : '';

            // ВАЖНО:
            // 1. Кнопка "Профиль" вызывает JS (модалку с паролем).
            // 2. Кнопка "Оставить отзыв" - это ТЕГ <a>. 
            //    Мы добавляем параметр ?teacher=..., чтобы на странице отзывов знать, кому он.
            const encodedName = encodeURIComponent(teacher.name);
            
            card.innerHTML = `
                <div class="img-container">
                    ${imageHtml}
                </div>
                <div class="card-name">${teacher.name}</div>
                
                <div class="overlay">
                    <!-- Кнопка 1: Профиль (требует пароль) -->
                    <div onclick="openPasswordModal('${teacher.name}')" class="view-btn">📂 Личный кабинет</div>
                    
                    <!-- Кнопка 2: Оставить отзыв (ГИПЕРССЫЛКА, как ты просил) -->
                    <a href="write-review.html?teacher=${encodedName}" class="review-link">✍️ Оставить отзыв</a>
                </div>
            `;
            list.appendChild(card);
        });
    </script>
